def slugtitle(title)
  title.gsub! /['`]/,""

  # @ --> at, and & --> and
  title.gsub! /\s*@\s*/, " at "
  title.gsub! /\s*&\s*/, " and "
  title.downcase.gsub! /\s*[^A-Za-z0-9\.\-]\s*/, '_'
end
